// --- LOGIC XỬ LÝ CHÍNH TRONG VERSION 1.16 --- async function generateQR() { const title = document.getElementById('qr-title').value.trim(); const url = document.getElementById('qr-url').value.trim(); const customShort = document.getElementById('qr-short-name').value.trim() .toLowerCase() .normalize("NFD").replace(/[\u0300-\u036f]/g, "") // Khử dấu tiếng Việt .replace(/\s+/g, '-') // Khoảng trắng thành gạch ngang .replace(/[^a-z0-9-]/g, ''); // Loại bỏ ký tự đặc biệt const oldId = document.getElementById('edit-id').value; // ID cũ khi bấm Sửa if (!title || !url) return alert("Vui lòng nhập Tên tài liệu và Link đích!"); if (customShort === "admin" || customShort === "index" || customShort === "go") { return alert("Tên rút gọn này hệ thống đã giữ chỗ, vui lòng chọn tên khác!"); } try { let finalId = customShort; // TRƯỜNG HỢP 1: TẠO MỚI if (!oldId) { if (!finalId) { // Nếu để trống tên rút gọn -> Tạo ID ngẫu nhiên const docRef = await db.collection('qrcodes').add({ uid: auth.currentUser.uid, title, url, date: firebase.firestore.Timestamp.now() }); finalId = docRef.id; } else { // Nếu tự nhập tên -> KIỂM TRA TRÙNG const check = await db.collection('qrcodes').doc(finalId).get(); if (check.exists) { return alert("❌ Tên '" + finalId + "' đã có người sử dụng. Vui lòng chọn tên khác!"); } await db.collection('qrcodes').doc(finalId).set({ uid: auth.currentUser.uid, title, url, date: firebase.firestore.Timestamp.now() }); } } // TRƯỜNG HỢP 2: ĐANG SỬA else { if (finalId && finalId !== oldId) { // Nếu người dùng đổi từ tên A sang tên B -> KIỂM TRA TÊN B CÓ AI DÙNG CHƯA const check = await db.collection('qrcodes').doc(finalId).get(); if (check.exists) { return alert("❌ Tên mới '" + finalId + "' đã bị trùng. Không thể đổi!"); } // Copy dữ liệu sang ID mới và xóa ID cũ await db.collection('qrcodes').doc(finalId).set({ uid: auth.currentUser.uid, title, url, date: firebase.firestore.Timestamp.now() }); await db.collection('qrcodes').doc(oldId).delete(); } else { // Nếu giữ nguyên tên cũ hoặc không nhập tên mới (vẫn dùng ID cũ) await db.collection('qrcodes').doc(oldId).update({ title, url }); finalId = oldId; } } // VẼ MÃ QR VỚI LINK MỚI const shortUrl = getShortUrl(finalId); qrcode.clear(); qrcode.makeCode(shortUrl); document.getElementById('qr-display').classList.remove('hidden'); alert("✨ Tuyệt vời! Link rút gọn của bạn đã sẵn sàng."); setTimeout(() => { const img = document.querySelector("#qrcode img").src; document.getElementById('dl-btn').onclick = () => { const a = document.createElement('a'); a.href = img; a.download = title + ".png"; a.click(); }; }, 600); } catch (e) { console.error(e); alert("Có lỗi xảy ra, vui lòng thử lại!"); } }
